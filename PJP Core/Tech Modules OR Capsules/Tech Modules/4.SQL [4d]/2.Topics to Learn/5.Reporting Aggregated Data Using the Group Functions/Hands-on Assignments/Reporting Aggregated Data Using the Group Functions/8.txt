SELECT MAX(E."TOTAL EMP") "TOTAL EMP", 
MAX(CASE WHEN V.YEAR = '1980' THEN V.TOTAL END) "EMP HIRED IN 1980", 
MAX(CASE WHEN V.YEAR = '1981' THEN V.TOTAL END) "EMP HIRED IN 1981", 
MAX(CASE WHEN V.YEAR = '1982' THEN V.TOTAL END) "EMP HIRED IN 1982"  
FROM (SELECT COUNT(*) "TOTAL EMP" FROM EMP) E,
(SELECT TO_CHAR(HIREDATE, 'YYYY') "YEAR", COUNT(*) "TOTAL" 
FROM EMP
WHERE TO_CHAR(HIREDATE, 'YYYY') IN ('1980', '1981', '1982')
GROUP BY TO_CHAR(HIREDATE, 'YYYY')) V;
